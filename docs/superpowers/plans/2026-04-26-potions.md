# Именованные зелья в /give — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Заменить 4 generic-пункта зелий в give.html на 20 именованных зелий с секцией выбора типа (обычное/бросаемое/длительное/стрела) и мощности (обычное/продлённое/усиленное II).

**Architecture:** Единственный файл `give.html`. Зелья хранятся как `potion:<effect>` в item-select. Хелпер `isPotion()` определяет тип. Новая функция `updatePotionModifiers()` управляет доступными опциями мощности. `buildComponents()` добавляет `potion_contents`, `generate()` подставляет реальный item ID из radio-кнопки типа.

**Tech Stack:** Vanilla HTML/CSS/JS, без сборки, без зависимостей.

function isPotion(id) {
  return (id || '').startsWith('potion:');
}

function updatePotionModifiers(itemId) {
  const effect = itemId.split(':')[1];
  const entry = POTIONS.find(p => p[0] === effect);
  const hasLong   = entry ? entry[2] : false;
  const hasStrong = entry ? entry[3] : false;

  const longLabel   = document.getElementById('potion-mod-long');
  const strongLabel = document.getElementById('potion-mod-strong');
  longLabel.style.display   = hasLong   ? '' : 'none';
  strongLabel.style.display = hasStrong ? '' : 'none';

  const checked = document.querySelector('input[name="potion-modifier"]:checked');
  if (checked && checked.value === 'long'   && !hasLong)   document.querySelector('input[name="potion-modifier"][value="normal"]').checked = true;
  if (checked && checked.value === 'strong' && !hasStrong) document.querySelector('input[name="potion-modifier"][value="normal"]').checked = true;
}
```

function restoreSnapshot(snap) {
  document.getElementById('item-select').value = snap.itemId;
  document.getElementById('target-preset').value = snap.target || '@s';
  document.getElementById('target-custom').value = snap.targetCustom || '';
  document.getElementById('item-count').value = snap.count || '1';
  onItemChange();
  if (isPotion(snap.itemId)) {
    const typeEl = document.querySelector(`input[name="potion-type"][value="${snap.potionType || 'potion'}"]`);
    if (typeEl) typeEl.checked = true;
    updatePotionModifiers(snap.itemId);
    const modEl = document.querySelector(`input[name="potion-modifier"][value="${snap.potionModifier || 'normal'}"]`);
    if (modEl && modEl.closest('label').style.display !== 'none') modEl.checked = true;
  }
  Object.entries(snap.fields || {}).forEach(([key, v]) => {
    const el = document.querySelector(`[data-field="${key}"]`);
    if (!el) return;
    if (el.type === 'checkbox') el.checked = (v === true || v === 'true');
    else el.value = v !== undefined ? v : '';
  });
  explosions = (snap.explosions || []).map(e => ({ ...e, id: nextExplosionId++ }));
  if (document.getElementById('item-select').value === 'firework_rocket') renderExplosions();
  generate();
}
```

function resetAll() {
  if (!confirm('Сбросить все параметры?')) return;
  document.getElementById('item-select').selectedIndex = 0;
  document.getElementById('item-search').value = '';
  document.getElementById('target-custom').value = '';

  explosions = [];
  renderExplosions();
  document.querySelectorAll('[data-field]').forEach(el => {
    if (el.type === 'checkbox') el.checked = false;
    else if (el.tagName === 'SELECT') el.selectedIndex = 0;
    else el.value = el.defaultValue || '';
  });
  initTotemEffectsRows();
  document.querySelector('input[name="potion-type"][value="potion"]').checked = true;
  document.querySelector('input[name="potion-modifier"][value="normal"]').checked = true;
  onItemChange();
}
